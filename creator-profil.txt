    // Micro-interaction for Follow Button
    const followBtn = document.querySelector('.soft-brutalism-button');
    let following = false;
    followBtn.addEventListener('click', () => {
      following = !following;
      followBtn.textContent = following ? 'Following' : 'Follow';
      followBtn.style.backgroundColor = following ? '#e5e2e1' : '#ffd1dc';
      followBtn.style.color = following ? '#4f4446' : '#7a5761';
    });

    // Micro-interaction for Like toggles
    const favoriteIcons = document.querySelectorAll('[data-icon="favorite"]');
    favoriteIcons.forEach(icon => {
      icon.parentElement.addEventListener('click', (e) => {
        e.stopPropagation();
        const currentFill = icon.style.fontVariationSettings.includes("'FILL' 1");
        icon.style.fontVariationSettings = `'FILL' ${currentFill ? 0 : 1}`;
        icon.classList.toggle('text-primary');
        icon.classList.toggle('text-on-surface-variant');
      });
    });
